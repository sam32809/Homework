#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	int x, y;
	int result = 0;
	cout << "input x=";
	cin >> x;
	cout << "input y=";
	cin >> y;

	while (x != 1)
	{
		if ((x&1)== 1)
		{
			result += y;
		}
		x = x >> 1;
		y = y << 1;
	}

	result += y;

	cout << "Result:" << result << endl;


	return 0;
}

//13번 문제

#include <cstdio>

int findMaxValue(int arr[],int len) {
	int max = 0;
	for (int i = 0; i < len; i++)
	{
		if (max < arr[i])
			max = arr[i];
	}
	return max;
}


int main()
{
	int arr[10] = { 3,24,83,12,34,7,53,17,26,51 };

	int maxVal = findMaxValue(arr,10);

	printf("배열의 최대값 = %d\n", maxVal);
	return 0;
}

//9번문제

#include <iostream>
using namespace std;

class Complex {
public:
	int re;
	int im;
	Complex(int a, int b)
	{
		re = a;
		im = b;
	}
	Complex(){}
	void print()
	{
		cout << "(" << re;
		if (im > 0) cout << "+";
		cout << im << "i)";
	}
	void add(Complex i, Complex j) 
	{
		re = (i.re + j.re);
		im = (i.im + j.im);
	}
};


int main()
{
	Complex a(7, 8);
	a.print();
	cout << "+";
	Complex b(2, 9);
	b.print();
	cout << "=";
	Complex c;
	c.add(a, b);
	c.print();
	cout << endl;

	Complex d(5, 3);
	d.print();
	cout << "+";
	Complex e(3, -4);
	e.print();
	cout << "=";
	Complex f;
	f.add(d, e);
	f.print();
	cout << endl;

	return 0;
}
